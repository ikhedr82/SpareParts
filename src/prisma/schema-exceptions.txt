// New models for exceptions, returns, and refunds
model DeliveryException {
  id              String                    @id @default(uuid())
  tenantId        String                    @map("tenant_id")
  tripStopId      String                    @map("trip_stop_id")
  exceptionType   DeliveryExceptionType     @map("exception_type")
  description     String
  reportedBy      String                    @map("reported_by")
  reportedAt      DateTime                  @default(now()) @map("reported_at")
  resolved        Boolean                   @default(false)
  resolutionType  String?                   @map("resolution_type")
  resolutionNotes String?                   @map("resolution_notes")
  resolvedBy      String?                   @map("resolved_by")
  resolvedAt      DateTime?                 @map("resolved_at")
  
  tenant          Tenant                    @relation(fields: [tenantId], references: [id])
  tripStop        TripStop                  @relation(fields: [tripStopId], references: [id])
  return          Return?
  
  @@index([tenantId])
  @@index([tripStopId])
  @@index([resolved])
  @@map("delivery_exceptions")
}

model Return {
  id                    String                @id @default(uuid())
  tenantId              String                @map("tenant_id")
  orderId               String                @map("order_id")
  deliveryExceptionId   String?               @unique @map("delivery_exception_id")
  returnNumber          String                @unique @map("return_number")
  reason                ReturnReason
  reasonNotes           String?               @map("reason_notes")
  status                ReturnStatus          @default(REQUESTED)
  requestedBy           String                @map("requested_by")
  requestedAt           DateTime              @default(now()) @map("requested_at")
  approvedBy            String?               @map("approved_by")
  approvedAt            DateTime?             @map("approved_at")
  receivedAt            DateTime?             @map("received_at")
  completedAt           DateTime?             @map("completed_at")
  
  tenant                Tenant                @relation(fields: [tenantId], references: [id])
  order                 Order                 @relation(fields: [orderId], references: [id])
  deliveryException     DeliveryException?    @relation(fields: [deliveryExceptionId], references: [id])
  returnItems           ReturnItem[]
  refund                Refund?
  
  @@index([tenantId])
  @@index([orderId])
  @@index([status])
  @@index([returnNumber])
  @@map("returns")
}

model ReturnItem {
  id                String      @id @default(uuid())
  returnId          String      @map("return_id")
  orderItemId       String      @map("order_item_id")
  quantity          Int
  condition         String?
  inspectionNotes   String?     @map("inspection_notes")
  restockable       Boolean     @default(true)
  
  return            Return      @relation(fields: [returnId], references: [id])
  orderItem         OrderItem   @relation(fields: [orderItemId], references: [id])
  
  @@index([returnId])
  @@index([orderItemId])
  @@map("return_items")
}

model Refund {
  id              String         @id @default(uuid())
  tenantId        String         @map("tenant_id")
  orderId         String         @map("order_id")
  returnId        String?        @unique @map("return_id")
  refundNumber    String         @unique @map("refund_number")
  amount          Decimal        @db.Decimal(10, 2)
  reason          String
  status          RefundStatus   @default(PENDING)
  paymentMethod   String?        @map("payment_method")
  createdBy       String         @map("created_by")
  createdAt       DateTime       @default(now()) @map("created_at")
  processedBy     String?        @map("processed_by")
  processedAt     DateTime?      @map("processed_at")
  cancelledBy     String?        @map("cancelled_by")
  cancelledAt     DateTime?      @map("cancelled_at")
  
  tenant          Tenant         @relation(fields: [tenantId], references: [id])
  order           Order          @relation(fields: [orderId], references: [id])
  return          Return?        @relation(fields: [returnId], references: [id])
  
  @@index([tenantId])
  @@index([orderId])
  @@index([status])
  @@index([refundNumber])
  @@map("refunds")
}
